DUENIOS_CHOFERES = pi dni(DUENIO) ⨝ pi dni(CHOFER)
DUENIOS_NO_CHOFERES = pi dni(DUENIO) - DUENIOS_CHOFERES

DUENIOS_CON_FIAT = pi dni(DUENIO ⨝ sigma marca = 'FIAT' (AUTO))
DUENIOS_CON_DIFERENTE_A_FIAT = pi dni(DUENIO ⨝ sigma marca <> 'FIAT' (AUTO))

DUENIOS_SOLO_FIAT = DUENIOS_CON_FIAT - DUENIOS_CON_DIFERENTE_A_FIAT

RESULTADO = (DUENIOS_NO_CHOFERES ∩ DUENIOS_SOLO_FIAT) ⨝ pi dni, nombre, telefono(DUENIO)
